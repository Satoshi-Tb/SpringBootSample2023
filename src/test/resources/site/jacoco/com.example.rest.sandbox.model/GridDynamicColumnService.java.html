	public List<RowDataV3> getRowDataV3(String id) {
		var base = getRowData(id);
		
		var list = new ArrayList<RowDataV3>();
		for (RowData src : base) {
			list.add(new RowDataV3(src.id(), src.category(), src.item(), src.detailItems()));
		}
		return list;
